# MCP Code Intelligence – Web UI Implementation Plan

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a React + Vite + TailwindCSS + Cytoscape.js web dashboard that lets users browse indexed repositories, search symbols, visualize dependency graphs, and monitor indexing status — all backed by a REST + SSE API on the existing Express server.

**Architecture:** The Express HTTP server (new — added to `App`) serves both the REST/SSE API and the static Vite build. The React SPA communicates with the backend via `fetch` and `EventSource`. There are 7 views: Dashboard, Repos, Symbol Search, File Browser, Symbol Detail, Dependency Graph, and Settings.

**Tech Stack:** React 18, Vite 5, TailwindCSS 3, Cytoscape.js 3, React Router 6, React Query (TanStack), Lucide React (icons), Express 4, Express SSE helper.

```typescript
import { Router } from 'express';
import type { RepoRegistry } from '../../registry.js';
import type { Indexer } from '../../indexer/indexer.js';
import { registerRepo } from '../mcp/tools/register-repo.js';
import { getRepoStats } from '../mcp/tools/get-repo-stats.js';
import type { Db } from '../../db/index.js';

export function reposRouter(registry: RepoRegistry, indexer: Indexer, db: Db): Router {
  const router = Router();

  router.get('/', (_req, res) => {
    res.json(registry.list());
  });

  router.post('/', (req, res) => {
    try {
      const { name, root_path, language } = req.body as { name: string; root_path: string; language?: string };
      const repo = registerRepo(registry, { name, rootPath: root_path, language });
      res.status(201).json(repo);
    } catch (e: unknown) {
      res.status(400).json({ error: String(e) });
    }
  });

  router.get('/:id/stats', (req, res) => {
    const stats = getRepoStats(db, req.params.id!);
    res.json(stats);
  });

  router.post('/:id/index', async (req, res) => {
    const repo = registry.getById(req.params.id!);
    if (!repo) { res.status(404).json({ error: 'Not found' }); return; }
    const result = await indexer.indexRepo(repo.id, repo.rootPath);
    res.json(result);
  });

  router.delete('/:id', (req, res) => {
    const repo = registry.getById(req.params.id!);
    if (!repo) { res.status(404).json({ error: 'Not found' }); return; }
    registry.remove(repo.id);
    res.status(204).send();
  });

  return router;
}
```

```typescript
import { Router } from 'express';
import type { Db } from '../../db/index.js';
import { searchSymbols } from '../../mcp/tools/search-symbols.js';
import { getSymbolDetail } from '../../mcp/tools/get-symbol-detail.js';
import { findReferences } from '../../mcp/tools/find-references.js';

export function symbolsRouter(db: Db): Router {
  const router = Router();

  router.get('/search', (req, res) => {
    const query = String(req.query['q'] ?? '');
    const repoId = req.query['repo_id'] ? String(req.query['repo_id']) : null;
    const kind = req.query['kind'] ? String(req.query['kind']) : null;
    const limit = req.query['limit'] ? Number(req.query['limit']) : 20;
    if (!query) { res.json([]); return; }
    res.json(searchSymbols(db, { query, repoId, kind, limit }));
  });

  router.get('/:id', (req, res) => {
    const detail = getSymbolDetail(db, req.params.id!);
    if (!detail) { res.status(404).json({ error: 'Not found' }); return; }
    res.json(detail);
  });

  router.get('/:id/references', (req, res) => {
    const detail = getSymbolDetail(db, req.params.id!);
    if (!detail) { res.status(404).json({ error: 'Not found' }); return; }
    const refs = findReferences(db, { symbolName: detail.name, repoId: null });
    res.json(refs);
  });

  return router;
}
```

```typescript
import { Router } from 'express';
import type { Response } from 'express';

type SseClient = { id: string; res: Response };

const clients = new Map<string, SseClient>();

export function eventsRouter(): Router {
  const router = Router();

  router.get('/', (req, res) => {
    const id = Math.random().toString(36).slice(2);
    res.writeHead(200, {
      'Content-Type': 'text/event-stream',
      'Cache-Control': 'no-cache',
      'Connection': 'keep-alive',
      'X-Accel-Buffering': 'no',
    });
    res.write('event: connected\ndata: {}\n\n');
    clients.set(id, { id, res });
    req.on('close', () => clients.delete(id));
  });

  return router;
}

export function broadcastEvent(event: string, data: unknown): void {
  const payload = `event: ${event}\ndata: ${JSON.stringify(data)}\n\n`;
  for (const client of clients.values()) {
    client.res.write(payload);
  }
}
```

```typescript
import express from 'express';
import { createServer, type Server } from 'node:http';
import type { Db } from '../db/index.js';
import type { RepoRegistry } from '../registry.js';
import type { Indexer } from '../indexer/indexer.js';
import { reposRouter } from './routes/repos.js';
import { symbolsRouter } from './routes/symbols.js';
import { filesRouter } from './routes/files.js';
import { eventsRouter } from './routes/events.js';

export function createHttpServer(opts: {
  db: Db;
  registry: RepoRegistry;
  indexer: Indexer;
  uiDistPath?: string;
}): Server {
  const app = express();
  app.use(express.json());

  // API routes
  app.use('/api/repos', reposRouter(opts.registry, opts.indexer, opts.db));
  app.use('/api/symbols', symbolsRouter(opts.db));
  app.use('/api/files', filesRouter(opts.db));
  app.use('/events', eventsRouter());

  // Serve built UI (production)
  if (opts.uiDistPath) {
    app.use(express.static(opts.uiDistPath));
    app.get('*', (_req, res) => {
      res.sendFile('index.html', { root: opts.uiDistPath });
    });
  }

  return createServer(app);
}
```

```typescript
const BASE = '/api';

export class ApiError extends Error {
  constructor(public status: number, message: string) {
    super(message);
    this.name = 'ApiError';
  }
}

async function request<T>(path: string, init?: RequestInit): Promise<T> {
  const res = await fetch(`${BASE}${path}`, {
    headers: { 'Content-Type': 'application/json', ...init?.headers },
    ...init,
  });
  if (!res.ok) {
    const body = await res.text().catch(() => '');
    throw new ApiError(res.status, body || res.statusText);
  }
  return res.json() as Promise<T>;
}

export const api = {
  repos: {
    list: () => request<import('./types.js').Repo[]>('/repos'),
    stats: (id: string) => request<import('./types.js').RepoStats>(`/repos/${id}/stats`),
    register: (body: { name: string; root_path: string; language?: string }) =>
      request<import('./types.js').Repo>('/repos', { method: 'POST', body: JSON.stringify(body) }),
    index: (id: string) => request<unknown>(`/repos/${id}/index`, { method: 'POST' }),
    remove: (id: string) => request<void>(`/repos/${id}`, { method: 'DELETE' }),
  },
  symbols: {
    search: (q: string, params?: { repoId?: string; kind?: string; limit?: number }) => {
      const sp = new URLSearchParams({ q });
      if (params?.repoId) sp.set('repo_id', params.repoId);
      if (params?.kind) sp.set('kind', params.kind);
      if (params?.limit) sp.set('limit', String(params.limit));
      return request<import('./types.js').Symbol[]>(`/symbols/search?${sp}`);
    },
    detail: (id: string) => request<import('./types.js').SymbolDetail>(`/symbols/${id}`),
    references: (id: string) => request<import('./types.js').Symbol[]>(`/symbols/${id}/references`),
  },
  files: {
    search: (q: string, repoId?: string) => {
      const sp = new URLSearchParams({ q });
      if (repoId) sp.set('repo_id', repoId);
      return request<import('./types.js').FileResult[]>(`/files/search?${sp}`);
    },
    symbols: (repoId: string, relPath: string) => {
      const sp = new URLSearchParams({ repo_id: repoId, rel_path: relPath });
      return request<import('./types.js').Symbol[]>(`/files/symbols?${sp}`);
    },
  },
};
```

```typescript
import { useEffect, useState } from 'react';

export type SseEvent = { type: string; data: unknown };

export function useSse(enabled = true) {
  const [events, setEvents] = useState<SseEvent[]>([]);

  useEffect(() => {
    if (!enabled) return;
    const es = new EventSource('/events');
    es.onmessage = (e) => {
      setEvents((prev) => [...prev.slice(-99), { type: 'message', data: JSON.parse(e.data as string) }]);
    };
    es.addEventListener('index_progress', (e) => {
      setEvents((prev) => [
        ...prev.slice(-99),
        { type: 'index_progress', data: JSON.parse((e as MessageEvent).data as string) },
```

```tsx
          style: { 'background-color': '#7c3aed' },
        },
        {
          selector: 'node[kind = "interface"]',
          style: { 'background-color': '#0d9488' },
        },
        {
          selector: 'edge',
          style: {
            'width': 1.5,
            'line-color': '#475569',
            'target-arrow-color': '#475569',
            'target-arrow-shape': 'triangle',
            'curve-style': 'bezier',
          },
        },
        {
          selector: ':selected',
          style: { 'border-width': 2, 'border-color': '#f59e0b' },
        },
      ],
      layout: { name: 'cose', padding: 30 } as cytoscape.LayoutOptions,
    });
    return () => cyRef.current?.destroy();
  }, []);

  const loadGraph = async () => {
    if (!query || !cyRef.current) return;
    const symbols: Symbol[] = await api.symbols.search(query, { repoId: repoId || undefined, limit: 50 });

    cyRef.current.elements().remove();

    const nodes = symbols.map((s) => ({
      data: { id: s.id, label: s.name, kind: s.kind, file: s.filePath },
    }));

    // Group by file → add file→symbol edges
    const fileNodes = [...new Set(symbols.map((s) => s.filePath))].map((f) => ({
      data: { id: `file:${f}`, label: f.split('/').at(-1) ?? f, kind: 'file' },
    }));

    const edges = symbols.map((s) => ({
      data: { id: `e:${s.id}`, source: `file:${s.filePath}`, target: s.id },
    }));

    cyRef.current.add([...fileNodes, ...nodes, ...edges]);
    cyRef.current.layout({ name: 'cose', padding: 40, animate: true } as cytoscape.LayoutOptions).run();
  };

  return (
    <div className="space-y-4 h-full flex flex-col">
      <h1 className="text-xl font-semibold text-slate-100">Dependency Graph</h1>

      <div className="flex gap-3">
        <input
          value={query}
          onChange={(e) => setQuery(e.target.value)}
          onKeyDown={(e) => e.key === 'Enter' && void loadGraph()}
          className="input flex-1"
          placeholder="Search symbols to visualize…"
        />
        <select value={repoId} onChange={(e) => setRepoId(e.target.value)} className="input w-48">
          <option value="">All repos</option>
          {repos.map((r) => <option key={r.id} value={r.id}>{r.name}</option>)}
        </select>
        <button onClick={() => void loadGraph()} className="btn-primary">Load Graph</button>
      </div>

      <div ref={containerRef} className="flex-1 min-h-96 bg-surface-950 rounded-lg border border-slate-700/50" />
    </div>
  );
}
```
